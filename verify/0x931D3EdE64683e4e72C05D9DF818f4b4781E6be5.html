This is the source used for deployment (stored alongside deploy scripts). The BlockScout backend's recompilation produces slightly different bytecode due to platform-specific differences in the compiler binary output. 

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

/**
 * @title CommunityTreasuryMinimal
 * @dev 最简化的国库合约（无 OpenZeppelin 依赖）
 * 
 * 功能：
 * 1. 接收 GDC（原生币）和 CGC（ERC20 代币）
 * 2. owner 可以提取资金（withdraw() / withdrawToken()）
 * 3. 后续将 owner 转为 TimelockController
 */
contract CommunityTreasuryMinimal {
    address public owner;
    bool public paused = false;
    
    event OwnershipTransferred(address indexed previousOwner, address indexed newOwner);
    event Paused(address account);
    event Unpaused(address account);
    event Withdraw(address indexed to, uint256 amount);
    event WithdrawToken(address indexed token, address indexed to, uint256 amount);
    
    constructor(address _owner) {
        owner = _owner;
    }
    
    modifier onlyOwner() {
        require(msg.sender == owner, "Treasury: caller is not the owner");
        _;
    }
    
    modifier whenNotPaused() {
        require(!paused, "Treasury: paused");
        _;
    }
    
    // 接收原生币（GDC）
    receive() external payable {}
    
    // 提取原生币（GDC）
    function withdraw(address to, uint256 amount) external onlyOwner whenNotPaused {
        require(to != address(0), "Treasury: zero address");
        require(amount <= address(this).balance, "Treasury: insufficient balance");
        payable(to).transfer(amount);
        emit Withdraw(to, amount);
    }
    
    // 提取 ERC20 代币（CGC）
    function withdrawToken(address token, address to, uint256 amount) external onlyOwner whenNotPaused {
        require(token != address(0), "Treasury: zero token address");
        require(to != address(0), "Treasury: zero address");
        // 直接调用 ERC20 转账（无 SafeERC20）
        (bool success, bytes memory data) = token.call(abi.encodeWithSignature("transfer(address,uint256)", to, amount));
        require(success && (data.length == 0 || abi.decode(data, (bool))), "Treasury: token transfer failed");
        emit WithdrawToken(token, to, amount);
    }
    
    // 转移 owner 权限
    function transferOwnership(address newOwner) external onlyOwner {
        require(newOwner != address(0), "Treasury: zero address");
        emit OwnershipTransferred(owner, newOwner);
        owner = newOwner;
    }
    
    // 暂停/恢复
    function pause() external onlyOwner {
        paused = true;
        emit Paused(msg.sender);
    }
    
    function unpause() external onlyOwner {
        paused = false;
        emit Unpaused(msg.sender);
    }
    
    // 查询余额（注意：getTokenBalance 需要 gas，因为它使用 call()）
    function getBalance() external view returns (uint256) {
        return address(this).balance;
    }
    
    function getTokenBalance(address token) external returns (uint256) {
        (bool success, bytes memory data) = token.call(abi.encodeWithSignature("balanceOf(address)", address(this)));
        require(success, "Treasury: balanceOf failed");
        return abi.decode(data, (uint256));
    }
}
